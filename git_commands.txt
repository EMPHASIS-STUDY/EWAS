#basic git commands
#author : Ayden Saffari

#config
git config --global user.name "Your Name"
git config --global user.email you@example.com

#add repo
git remote add origin https://github.com/asaffa/EMPHASIS.git

#get latest
git pull

#add file in current directory 
git add myfile.txt

#add all files in current directory
git add -A .

#remove file in current directory 
git rm myfile.txt

#commit changes
git commit -a -m 'meffil latest & analysis plan & git commands'

#push changes to repo master branch
git push -u origin master

#reset local repo (warning: will remove all local files that havent been added)
git fetch --all
git reset --hard origin/master
